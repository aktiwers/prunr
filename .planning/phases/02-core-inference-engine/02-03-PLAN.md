```rust
use crate::types::{CoreError, ModelKind};
use ort::{
    execution_providers::CPUExecutionProvider,
    session::{Session, GraphOptimizationLevel},
};

/// Trait for inference backends. Implemented by OrtEngine.
/// Send + Sync required because worker threads own the engine instance.
pub trait InferenceEngine: Send + Sync {
    /// Returns the name of the active execution provider (e.g., "CUDA", "CoreML", "CPU").
    fn active_provider(&self) -> &str;
}

/// ORT-backed inference engine. Holds one Session per model selection.
/// Create once per model; reuse across all images — never instantiate per-image.
pub struct OrtEngine {
    session: Session,
    provider_name: String,
}

impl OrtEngine {
    /// Create a new OrtEngine for the given model.
    ///
    /// - `model`: Which ONNX model to load (Silueta = ~4MB fast, U2net = ~170MB quality)
    /// - `intra_threads`: ORT intra-op thread count. For batch use: num_cpus / rayon_workers.
    ///   Pass 1 for single-image use; the thread count is set at session creation time.
    ///
    /// Execution providers are registered in priority order: CUDA → CoreML → DirectML → CPU.
    /// ORT silently selects the first available EP. Call active_provider() after creation
    /// to confirm which EP was selected.
    pub fn new(model: ModelKind, intra_threads: usize) -> Result<Self, CoreError> {
        let model_bytes: &[u8] = match model {
            ModelKind::Silueta => {
                #[cfg(feature = "dev-models")]
                { return Self::new_from_bytes(&bgprunr_models::silueta_bytes(), intra_threads); }
                #[cfg(not(feature = "dev-models"))]
                bgprunr_models::SILUETA_BYTES
            }
            ModelKind::U2net => {
                #[cfg(feature = "dev-models")]
                { return Self::new_from_bytes(&bgprunr_models::u2net_bytes(), intra_threads); }
                #[cfg(not(feature = "dev-models"))]
                bgprunr_models::U2NET_BYTES
            }
        };

        Self::new_from_bytes(model_bytes, intra_threads)
    }

    fn new_from_bytes(model_bytes: &[u8], intra_threads: usize) -> Result<Self, CoreError> {
        let session = Session::builder()?
            .with_optimization_level(GraphOptimizationLevel::Level3)?
            .with_intra_op_threads(intra_threads.max(1) as i16)?
            .with_execution_providers([
                #[cfg(all(feature = "cuda", not(target_os = "macos")))]
                ort::execution_providers::CUDAExecutionProvider::default().build(),
                #[cfg(target_os = "macos")]
                ort::execution_providers::CoreMLExecutionProvider::default().build(),
                #[cfg(windows)]
                ort::execution_providers::DirectMLExecutionProvider::default().build(),
                CPUExecutionProvider::default().build(),
            ])?
            .commit_from_memory(model_bytes)?;

        // Determine which EP ORT selected by inspecting the session.
        // ORT does not expose a direct "active EP" API; we infer it from available providers.
        // The session's execution_providers list reflects what was requested — but the
        // first non-CPU provider in the registered list that ORT accepted is what's active.
        // For correctness we log CPU as fallback and rely on GPU utilization monitoring
        // to confirm GPU use. This matches rembg's behavior.
        let provider_name = Self::detect_active_provider();

        Ok(Self { session, provider_name })
    }

    /// Infer the active provider name. ORT 2.0-rc.12 does not expose a direct query API,
    /// so we check compile-time feature flags and runtime availability signals.
    fn detect_active_provider() -> String {
        // On macOS, if CoreML feature is compiled, assume CoreML is active when available.
        // On Linux/Windows with CUDA, assume CUDA is active if CUDA feature is compiled.
        // Real-world: ORT will log the selected EP to stderr at session init — visible in
        // development. For the API, return a best-effort string.
        #[cfg(target_os = "macos")]
        { return "CoreML".to_string(); }

        #[cfg(all(feature = "cuda", not(target_os = "macos")))]
        { return "CUDA".to_string(); }

        #[cfg(windows)]
        { return "DirectML".to_string(); }

        "CPU".to_string()
    }

    /// Access the underlying ORT Session for inference.
    /// Used by pipeline.rs — not part of the public trait API.
    pub(crate) fn session(&self) -> &Session {
        &self.session
    }
}

impl InferenceEngine for OrtEngine {
    fn active_provider(&self) -> &str {
        &self.provider_name
    }
}
```

```rust
use crate::{
    engine::OrtEngine,
    formats::{load_image_from_bytes, check_large_image, encode_rgba_png},
    preprocess::preprocess,
    postprocess::postprocess,
    types::{CoreError, ProcessResult, ProgressStage},
};
use ort::inputs;

/// Process a single image: remove background and return a transparent PNG.
///
/// # Arguments
/// - `img_bytes`: Raw image bytes (PNG, JPEG, WebP, or BMP)
/// - `engine`: OrtEngine with a loaded session. Create once, reuse across images.
/// - `progress`: Optional progress callback. Called at each pipeline stage.
///   Signature: `|stage: ProgressStage, pct: f32|`
///   Stages (in order): Decode(0.0) → Resize(0.2) → Normalize(0.4) → Infer(0.5)
///   → Postprocess(0.8) → Alpha(0.95) → done reported at return (1.0 not called explicitly)
///
/// # Errors
/// - `CoreError::LargeImage` if the image exceeds 8000px in either dimension
/// - `CoreError::ImageFormat` if the bytes cannot be decoded as a supported format
/// - `CoreError::Inference` if ORT inference fails
pub fn process_image<F>(
    img_bytes: &[u8],
    engine: &OrtEngine,
    progress: Option<F>,
) -> Result<ProcessResult, CoreError>
where
    F: Fn(ProgressStage, f32),
{
    let report = |stage: ProgressStage, pct: f32| {
        if let Some(ref cb) = progress {
            cb(stage, pct);
        }
    };

    // Stage 1: Decode
    report(ProgressStage::Decode, 0.0);
    let img = load_image_from_bytes(img_bytes)?;

    // Stage 1.5: Large image guard — return error before allocating a huge tensor
    if let Some(err) = check_large_image(&img) {
        return Err(err);
    }

    // Stage 2: Resize (happens inside preprocess)
    report(ProgressStage::Resize, 0.2);

    // Stage 3: Normalize (happens inside preprocess, reported before the call)
    report(ProgressStage::Normalize, 0.4);
    let input_tensor = preprocess(&img);

    // Stage 4: Inference
    report(ProgressStage::Infer, 0.5);
    let session = engine.session();
    let input_name = &session.inputs[0].name;
    let outputs = session.run(inputs![input_name => &input_tensor]?)?;
    let raw_output = outputs[0]
        .try_extract_tensor::<f32>()?
        .into_owned()
        .into_dimensionality::<ndarray::Ix4>()
        .map_err(|e| CoreError::Inference(format!("Output reshape error: {e}")))?;

    // Stage 5: Postprocess (min-max normalize → grayscale mask → resize to original dims)
    report(ProgressStage::Postprocess, 0.8);
    let rgba_image = postprocess(raw_output.view(), &img);

    // Stage 6: Alpha merge + PNG encode
    report(ProgressStage::Alpha, 0.95);
    let rgba_bytes = encode_rgba_png(&rgba_image)?;

    Ok(ProcessResult {
        rgba_bytes,
        active_provider: engine.active_provider().to_string(),
    })
}
```
