Create `.github/workflows/release.yml`. This workflow triggers on version tags and uses cargo-dist to produce per-platform binary archives.

**Important:** cargo-dist typically generates this file via `cargo dist init`. For Phase 1, create it manually to match cargo-dist 0.31.0's expected format. The key requirement is that it triggers on tags and runs `cargo dist build` for each platform target.

```yaml
name: Release

on:
  push:
    tags:
      - "v[0-9]*"

permissions:
  contents: write

jobs:
  plan:
    name: Plan release
    runs-on: ubuntu-latest
    outputs:
      val: ${{ steps.plan.outputs.manifest }}
      tag: ${{ !startsWith(github.ref, 'refs/tags/') && 'dry-run' || '' }}
    env:
      GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    steps:
      - uses: actions/checkout@v4
        with:
          submodules: recursive
      - name: Install cargo-dist
        run: |
          curl --proto '=https' --tlsv1.2 -LsSf \
            https://github.com/axodotdev/cargo-dist/releases/download/v0.31.0/cargo-dist-installer.sh \
            | sh
      - id: plan
        run: cargo dist plan --output-format=json > dist-manifest.json && echo "manifest=$(cat dist-manifest.json)" >> "$GITHUB_OUTPUT"
      - name: Store manifest
        uses: actions/upload-artifact@v4
        with:
          name: dist-manifest.json
          path: dist-manifest.json

  build:
    name: Build (${{ matrix.target }})
    needs: plan
    runs-on: ${{ matrix.os }}
    strategy:
      fail-fast: false
      matrix:
        include:
          - os: ubuntu-latest
            target: x86_64-unknown-linux-gnu
          - os: macos-13
            target: x86_64-apple-darwin
          - os: macos-14
            target: aarch64-apple-darwin
          - os: windows-latest
            target: x86_64-pc-windows-msvc
    env:
      GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    steps:
      - uses: actions/checkout@v4

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.target }}

      - name: Cache Cargo artifacts
        uses: Swatinem/rust-cache@v2
        with:
          key: release-${{ matrix.target }}

      - name: Restore model cache
        id: cache-models
        uses: actions/cache@v4
        with:
          path: models/
          key: models-${{ hashFiles('xtask/src/main.rs') }}

      - name: Fetch models (cache miss)
        if: steps.cache-models.outputs.cache-hit != 'true'
        run: cargo xtask fetch-models

      - name: Install cargo-dist
        run: |
          curl --proto '=https' --tlsv1.2 -LsSf \
            https://github.com/axodotdev/cargo-dist/releases/download/v0.31.0/cargo-dist-installer.sh \
            | sh

      - name: Build release binary
        run: cargo dist build --target=${{ matrix.target }} --output-format=json

      - name: Upload artifacts
        uses: actions/upload-artifact@v4
        with:
          name: artifacts-${{ matrix.target }}
          path: |
            target/distrib/

  publish:
    name: Publish GitHub Release
    needs: [plan, build]
    runs-on: ubuntu-latest
    env:
      GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    steps:
      - uses: actions/checkout@v4

      - name: Download dist manifest
        uses: actions/download-artifact@v4
        with:
          name: dist-manifest.json

      - name: Download all build artifacts
        uses: actions/download-artifact@v4
        with:
          pattern: artifacts-*
          merge-multiple: true
          path: dist-artifacts/

      - name: Install cargo-dist
        run: |
          curl --proto '=https' --tlsv1.2 -LsSf \
            https://github.com/axodotdev/cargo-dist/releases/download/v0.31.0/cargo-dist-installer.sh \
            | sh

      - name: Create GitHub Release
        run: cargo dist publish --output-format=json
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

Note: cargo-dist's exact workflow format varies by version. If `cargo dist init` is run interactively on the developer's machine with cargo-dist 0.31.0 installed, it will regenerate this file with the correct format. The above is a functional approximation. The CRITICAL requirement is that it triggers on version tags and calls `cargo dist build` per target.
